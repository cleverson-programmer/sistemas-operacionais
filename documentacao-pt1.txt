Problem (BBP), consiste em um desafio de sincronização de duas ou mais threads
(tarefas) concorrentes que têm acesso a um mesmo recurso do programa que está

package ProdutorConsumidor;

public class Produtor extends Thread {
    private Buffer buf;
    private int numero;

    public Produtor(Buffer buf, int numero) {
        this.buf = buf;
        this.numero = numero;
    }

    public void run() {
        for (int i = 0; i < 5; i++) { // armazena 5 valores no buffer
            buf.put(i);
            System.out.println("Produtor #" + numero + " put: " + i);
            try {
                sleep((int) (Math.random() * 100));
            } catch (InterruptedException e) {
                // tratamento opcional
            }
        }
    }
}
